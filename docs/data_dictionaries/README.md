# SG data vis - Data Dictionaries

adate (datetime )  - YYYY-MM-DD hh:mm:ss

machinename	(str) - Gantry 1|Gantry 2| Gantry 3| Gantry 4

energy (float, MeV) - 240 - 70

device	(str)  - XRV-3000 | XRV-4000

gantry angle (float) - 0, 90, 180, 270

spot (str) - Top-Top-Left .... Bottom-Bottom-left

x-pos (float) - position

y-pos (float) - position

hor_rt_gradient	 (float) - 80% -20% gradient for the horizontal right profile.

hor_lt_gradient	 (float) - 80% -20% gradient for the horizontal left profile.

hor_fwhm (float) - full width half max of horizontal profile

vert_rt_gradient    (float) - 80% -20% gradient for the vertical right profile.

vert_lt_gradient	(float) - 80% -20% gradient for the vertical left profile.

vert_fwhm	(float) - full width half max of vertical profile

bltr_rt_gradient	(float) - 80% -20% gradient for the bltr right profile.

bltr_lt_gradient	(float) - 80% -20% gradient for the bltr left profile.

bltr_fwhm	(float) - full width half max of bltr profile

tlbr_rt_gradient	(float) - 80% -20% gradient for the tlbr right profile.

tlbr_lt_gradient	(float) - 80% -20% gradient for the tlbr left profile.

tlbr_fwhm   (float) - full width half max of tlbr profile
